// Função que retorna o fatorial de um número inteiro:
FUNCTION INT _factorial (INT x){
   IF ( x = 0 || x = 1){
      RETURN 1;
   }
   ELSE{
      RETURN (x * _factorial(x - 1));
   }
   RETURN 0;
}


// Função que retorna o maior inteiro:
FUNCTION INT _max(INT x, INT y) {
   IF ( x > y){
       RETURN x;
   }
   ELSE{
       RETURN y;
   }
   RETURN 0;
}
